<script>
    const tg = window.Telegram.WebApp;
    tg.ready();
    tg.expand();

    const fileInput = document.getElementById('file-input');
    const preview = document.getElementById('preview');
    const mapContainer = document.getElementById('map-container');
    const dataContainer = document.getElementById('data-container');
    const metaTable = document.getElementById('meta-table-body');
    const coordsText = document.getElementById('coords-text');

    // Функция конвертации координат в десятичный формат
    function toDecimal(number, ref) {
        if (!number) return null;
        const decimal = number[0] + (number[1] / 60) + (number[2] / 3600);
        return (ref === "S" || ref === "W") ? decimal * -1 : decimal;
    }

    fileInput.addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (!file) return;

        // Очистка предыдущих результатов
        metaTable.innerHTML = '';
        mapContainer.style.display = 'none';
        dataContainer.classList.add('hidden');

        // Показ превью
        const reader = new FileReader();
        reader.onload = (event) => {
            preview.src = event.target.result;
            preview.style.display = 'block';
        };
        reader.readAsDataURL(file);

        // Работа с EXIF
        EXIF.getData(file, function() {
            const allTags = EXIF.getAllTags(this);
            
            if (Object.keys(allTags).length === 0) {
                tg.showAlert("Метаданные не найдены. Возможно, Telegram или телефон удалил их при сохранении.");
                return;
            }

            // Обработка GPS
            const lat = toDecimal(allTags.GPSLatitude, allTags.GPSLatitudeRef);
            const lon = toDecimal(allTags.GPSLongitude, allTags.GPSLongitudeRef);

            if (lat && lon) {
                mapContainer.style.display = 'block';
                const latFixed = lat.toFixed(6);
                const lonFixed = lon.toFixed(6);
                coordsText.innerText = `${latFixed}, ${lonFixed}`;
                document.getElementById('google-maps').href = `https://www.google.com/maps?q=${latFixed},${lonFixed}`;
                document.getElementById('yandex-maps').href = `https://yandex.ru/maps/?text=${latFixed},${lonFixed}`;
            }

            // Заполнение таблицы данными
            for (let tag in allTags) {
                if (tag === 'thumbnail' || typeof allTags[tag] === 'object' && !Array.isArray(allTags[tag])) continue;
                
                const row = document.createElement('tr');
                row.innerHTML = `<td>${tag}</td><td>${allTags[tag]}</td>`;
                metaTable.appendChild(row);
            }
            dataContainer.classList.remove('hidden');
        });
    });
</script>
